from hermes_agent import Agent

jarvis = Agent(
    model="deepseek/deepseek-v4-pro",
    provider="openrouter",
    tools=["terminal","browser","computer_use",
           "web_search","code_exec","image_gen",
           "cron","xurl","memory","delegate",
           "voice_tts","process"],
    skills="autoload",
    memory=True,
    creator="@kubiyaro"
